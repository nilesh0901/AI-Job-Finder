## Task v2.1-A — Fix Add Job URL Scraper

**File:** `backend-py/scraper.py`

### Problem
`POST /scrape` fails silently or returns empty fields for most URLs because:
1. No `User-Agent` header → sites reject the request or return a login wall
2. No redirect following on LinkedIn/Indeed (they 301 to login pages)
3. `og:description` is grabbed instead of the actual job description body
4. No timeout → request hangs

### Fix

```python
# backend-py/scraper.py — replace the existing scrapeJob / scrape_job function

import httpx
from bs4 import BeautifulSoup

HEADERS = {
    "User-Agent": (
        "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
        "AppleWebKit/537.36 (KHTML, like Gecko) "
        "Chrome/124.0.0.0 Safari/537.36"
    ),
    "Accept-Language": "en-US,en;q=0.9",
    "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8",
}

async def scrape_job(url: str) -> dict:
    """
    Returns dict: { title, company, description, location }
    Raises httpx.HTTPError on network failure.
    Raises ValueError if page returns a login wall or empty body.
    """
    async with httpx.AsyncClient(
        headers=HEADERS,
        follow_redirects=True,
        timeout=15.0,
    ) as client:
        r = await client.get(url)
        r.raise_for_status()

    soup = BeautifulSoup(r.text, "html.parser")

    # ── Title ────────────────────────────────────────────────────────────────
    title = (
        _meta(soup, "og:title")
        or _first_text(soup, ["h1", "h2"])
        or ""
    ).strip()

    # Strip site suffixes like " | LinkedIn" or " - Indeed"
    for sep in [" | ", " - ", " – ", " — "]:
        if sep in title:
            title = title.split(sep)[0].strip()
            break

    # ── Company ──────────────────────────────────────────────────────────────
    company = (
        _meta(soup, "og:site_name")
        or _first_text(soup, [
            '[data-company-name]',
            '.company-name', '.employer-name',
            '[class*="company"]', '[class*="employer"]',
        ])
        or ""
    ).strip()

    # ── Description ──────────────────────────────────────────────────────────
    desc_el = (
        soup.select_one('[class*="description"]')
        or soup.select_one('[class*="job-desc"]')
        or soup.select_one('[id*="description"]')
        or soup.select_one("article")
        or soup.select_one("main")
    )

    if desc_el:
        description = desc_el.get_text(separator="\n", strip=True)
    else:
        # Fallback: strip nav/footer/script and grab body text
        for tag in soup(["nav", "footer", "script", "style", "header"]):
            tag.decompose()
        description = soup.get_text(separator="\n", strip=True)

    # Truncate to 6000 chars — enough for AI, avoids token blowout
    description = description[:6000]

    # Detect login walls
    wall_signals = ["sign in to", "log in to", "create an account", "join linkedin"]
    if any(s in description[:500].lower() for s in wall_signals):
        raise ValueError(
            "Job site requires login to view this posting. "
            "Copy-paste the job description manually into the notes field."
        )

    # ── Location ─────────────────────────────────────────────────────────────
    location = (
        _first_text(soup, [
            '[class*="location"]', '[class*="workplace"]',
            '[data-location]', '[class*="city"]',
        ])
        or ""
    ).strip()

    return {
        "title": title or "Job Title",
        "company": company or "Company",
        "description": description,
        "location": location,
    }


def _meta(soup, property_: str) -> str:
    el = soup.find("meta", property=property_) or soup.find("meta", attrs={"name": property_})
    return el["content"].strip() if el and el.get("content") else ""


def _first_text(soup, selectors: list) -> str:
    for sel in selectors:
        el = soup.select_one(sel)
        if el:
            return el.get_text(strip=True)
    return ""
```

## Task v2.1-D — Job Suggestions Agent

**File:** `backend-py/suggestions.py` (create new)

```python
"""
suggestions.py — Job Suggestions Agent for AI Job Finder v2.1

Fetches jobs from free public APIs (RemoteOK, Arbeitnow, HN Algolia),
filters by user preferences, validates URLs, deduplicates against existing
board jobs, and inserts up to 10 suggestions per source into the jobs table.

Uses the SUPABASE_SERVICE_ROLE_KEY (server-side only — never expose to browser).
RLS is bypassed for inserts; suggestions land in the user's board with
is_suggestion=true and status='wishlist'.
"""

import asyncio
import os
from datetime import datetime, timezone, timedelta
from typing import Optional
import httpx
from supabase import create_client, Client

REMOTEOK_URL = "https://remoteok.com/api"
ARBEITNOW_URL = "https://www.arbeitnow.com/api/job-board-api"
HN_ALGOLIA_URL = "https://hn.algolia.com/api/v1/search"

HEADERS = {
    "User-Agent": "AIJobFinder/2.1 (job suggestions agent; contact nilesh.k0901@gmail.com)",
    "Accept": "application/json",
}

MAX_PER_SOURCE = 10


def _get_supabase() -> Client:
    url = os.getenv("SUPABASE_URL")
    key = os.getenv("SUPABASE_SERVICE_ROLE_KEY")
    if not url or not key:
        raise RuntimeError("SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY not set")
    return create_client(url, key)


async def _validate_url(client: httpx.AsyncClient, url: str) -> bool:
    """HEAD-check a URL. Returns True if the job page is reachable (2xx or 3xx)."""
    try:
        r = await client.head(url, timeout=8.0, follow_redirects=True)
        return r.status_code < 400
    except Exception:
        return False


def _cutoff_dt(freshness_days: int) -> datetime:
    return datetime.now(timezone.utc) - timedelta(days=freshness_days)


async def _fetch_remoteok(keywords: list[str], freshness_days: int) -> list[dict]:
    """Fetch jobs from RemoteOK public API."""
    results = []
    cutoff = _cutoff_dt(freshness_days)

    async with httpx.AsyncClient(headers=HEADERS, timeout=20.0) as client:
        try:
            r = await client.get(REMOTEOK_URL)
            r.raise_for_status()
            jobs = r.json()
            # First item is metadata notice, skip it
            if jobs and isinstance(jobs[0], dict) and "legal" in jobs[0]:
                jobs = jobs[1:]
        except Exception as e:
            print(f"[suggestions] RemoteOK fetch failed: {e}")
            return []

        kw_lower = [k.lower() for k in keywords]

        for job in jobs:
            if not isinstance(job, dict):
                continue

            # Date filter
            epoch = job.get("epoch", 0)
            if epoch:
                posted = datetime.fromtimestamp(epoch, tz=timezone.utc)
                if posted < cutoff:
                    continue

            # Keyword relevance
            searchable = f"{job.get('position','')} {job.get('description','')} {' '.join(job.get('tags', []))}".lower()
            if not any(kw in searchable for kw in kw_lower):
                continue

            url = job.get("url", "")
            if not url:
                continue

            # Validate URL
            valid = await _validate_url(client, url)
            if not valid:
                continue

            results.append({
                "title": job.get("position", "")[:200],
                "company": job.get("company", "")[:200],
                "url": url,
                "location": "Remote",
                "source": "RemoteOK",
                "notes": f"Source: RemoteOK | Tags: {', '.join(job.get('tags', [])[:5])}",
            })

            if len(results) >= MAX_PER_SOURCE:
                break

    return results


async def _fetch_arbeitnow(keywords: list[str], freshness_days: int) -> list[dict]:
    """Fetch jobs from Arbeitnow public API (no auth required)."""
    results = []
    cutoff = _cutoff_dt(freshness_days)

    async with httpx.AsyncClient(headers=HEADERS, timeout=20.0) as client:
        try:
            r = await client.get(ARBEITNOW_URL, params={"page": 1})
            r.raise_for_status()
            data = r.json()
            jobs = data.get("data", [])
        except Exception as e:
            print(f"[suggestions] Arbeitnow fetch failed: {e}")
            return []

        kw_lower = [k.lower() for k in keywords]

        for job in jobs:
            # Date filter
            created_at = job.get("created_at", "")
            if created_at:
                try:
                    posted = datetime.fromisoformat(created_at.replace("Z", "+00:00"))
                    if posted < cutoff:
                        continue
                except Exception:
                    pass

            # Keyword relevance
            searchable = f"{job.get('title','')} {job.get('description','')} {' '.join(job.get('tags', []))}".lower()
            if not any(kw in searchable for kw in kw_lower):
                continue

            url = job.get("url", "")
            if not url:
                continue

            valid = await _validate_url(client, url)
            if not valid:
                continue

            results.append({
                "title": job.get("title", "")[:200],
                "company": job.get("company_name", "")[:200],
                "url": url,
                "location": job.get("location", "Remote")[:200],
                "source": "Arbeitnow",
                "notes": f"Source: Arbeitnow | Tags: {', '.join(job.get('tags', [])[:5])}",
            })

            if len(results) >= MAX_PER_SOURCE:
                break

    return results


async def _fetch_hn(keywords: list[str], freshness_days: int) -> list[dict]:
    """Fetch 'Who is Hiring' posts from HN Algolia search."""
    results = []
    cutoff = _cutoff_dt(freshness_days)
    query = " OR ".join(keywords[:4]) + " hiring"  # HN search query

    async with httpx.AsyncClient(headers=HEADERS, timeout=20.0) as client:
        try:
            r = await client.get(HN_ALGOLIA_URL, params={
                "query": query,
                "tags": "comment,story",
                "numericFilters": f"created_at_i>{int(cutoff.timestamp())}",
                "hitsPerPage": 30,
            })
            r.raise_for_status()
            data = r.json()
            hits = data.get("hits", [])
        except Exception as e:
            print(f"[suggestions] HN Algolia fetch failed: {e}")
            return []

        for hit in hits:
            text = (hit.get("comment_text") or hit.get("title") or "").lower()
            # Must look like a hiring post with a URL
            if "hiring" not in text and "looking for" not in text:
                continue

            # Find a URL in the text
            import re
            urls = re.findall(r'https?://[^\s<>"\']+', text)
            url = urls[0] if urls else f"https://news.ycombinator.com/item?id={hit.get('objectID','')}"

            # Extract title — take first line of text or the HN story title
            title_raw = hit.get("title") or text.split("\n")[0][:150]
            title = (title_raw or "HN Job Posting")[:200]

            valid = await _validate_url(client, url)
            if not valid:
                continue

            results.append({
                "title": title,
                "company": hit.get("author", "")[:200],
                "url": url,
                "location": "Remote / Various",
                "source": "HackerNews",
                "notes": f"Source: HackerNews | {hit.get('created_at','')}",
            })

            if len(results) >= MAX_PER_SOURCE:
                break

    return results


async def _get_existing_urls(supabase: Client, user_id: str) -> set[str]:
    """Return the set of job URLs already on this user's board."""
    response = supabase.table("jobs").select("url").eq("user_id", user_id).execute()
    return {row["url"] for row in (response.data or []) if row.get("url")}


async def refresh_suggestions(user_id: str) -> dict:
    """
    Main entry point called by POST /suggestions/refresh.
    
    1. Load user profile (keywords = field + tech_stack + custom_skills)
    2. Fan out to RemoteOK, Arbeitnow, HN Algolia in parallel
    3. Dedup against existing board jobs
    4. Insert up to 10 per source as wishlist suggestions
    
    Returns { inserted: int, sources: dict }
    """
    supabase = _get_supabase()

    # Load user profile
    profile_resp = supabase.table("user_profiles").select("*").eq("user_id", user_id).maybe_single().execute()
    profile = profile_resp.data or {}

    # Build keyword list from profile
    keywords = []
    if profile.get("field"):
        keywords.append(profile["field"])
    if profile.get("domain"):
        keywords.append(profile["domain"])
    keywords.extend(profile.get("tech_stack") or [])
    keywords.extend(profile.get("custom_skills") or [])
    # Deduplicate, keep max 15 keywords
    keywords = list(dict.fromkeys(k for k in keywords if k))[:15]

    if not keywords:
        keywords = ["software engineer", "developer"]  # safe fallback

    freshness_days = profile.get("job_freshness_days") or 7

    # Fetch from all sources in parallel
    remote_ok_jobs, arbeitnow_jobs, hn_jobs = await asyncio.gather(
        _fetch_remoteok(keywords, freshness_days),
        _fetch_arbeitnow(keywords, freshness_days),
        _fetch_hn(keywords, freshness_days),
    )

    # Dedup against existing board
    existing_urls = await _get_existing_urls(supabase, user_id)

    all_candidates = remote_ok_jobs + arbeitnow_jobs + hn_jobs
    new_jobs = [j for j in all_candidates if j["url"] not in existing_urls]

    if not new_jobs:
        return {"inserted": 0, "sources": {"remoteok": 0, "arbeitnow": 0, "hackernews": 0}}

    # Build insert rows
    now = datetime.now(timezone.utc).isoformat()
    rows = [
        {
            "user_id": user_id,
            "title": j["title"],
            "company": j["company"],
            "url": j["url"],
            "location": j.get("location", ""),
            "status": "wishlist",
            "notes": j.get("notes", ""),
            "source": j.get("source", ""),
            "is_suggestion": True,
            "added_at": now,
        }
        for j in new_jobs
    ]

    supabase.table("jobs").insert(rows).execute()

    source_counts = {
        "remoteok": sum(1 for j in new_jobs if j.get("source") == "RemoteOK"),
        "arbeitnow": sum(1 for j in new_jobs if j.get("source") == "Arbeitnow"),
        "hackernews": sum(1 for j in new_jobs if j.get("source") == "HackerNews"),
    }

    return {"inserted": len(rows), "sources": source_counts}
```
